# Add postgres system user
/usr/sbin/useradd postgres
passwd postgres
(new pass is 'xxxxx')

# Setup the database to run
cd /usr/local/pgsql_8_2_6/
mkdir data
chown postgres data
su postgres
./bin/initdb ./data
./bin/postmaster -D ./data/ > ./data/logfile.txt 2>&1 &

# Create a user
./bin/createuser xxxxxxxx
